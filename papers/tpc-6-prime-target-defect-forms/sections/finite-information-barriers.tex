\section{Finite-information barriers}\label{sec:finite-barriers}

The exact character formulas above describe everything visible at a fixed
finite congruence resolution.  This section proves two correspondingly
finite limitations.  The first concerns nonnegative pointwise minorants.  The
second is a nullspace statement for all observables that factor through a
fixed CRT pushforward.

\subsection{Periodic prime minorants collapse}

Call $F:\Z^2\to\R$ \emph{$Q$-local} if
\begin{equation}\label{eq:q-local-observable}
 F(m+Q,n)=F(m,n+Q)=F(m,n)
\end{equation}
for all $m,n$.  Such an observable is a function on
$(\Z/Q\Z)^2$.  Every fixed-$Q$ factor-residue statistic, finite character
moment whose modulus divides $Q$, and target-rough kernel at a squarefree
modulus dividing $Q$ is $Q$-local.

\begin{theorem}[Periodic-minorant collapse]\label{thm:periodic-minorant-collapse}
Fix integers $Q\ge1$ and $h\ne0$.  Let $F:\Z^2\to[0,\infty)$ be $Q$-local.
If, for all sufficiently large positive $m,n$,
\begin{equation}\label{eq:prime-minorant-condition}
 F(m,n)\le\one_{\Pp}(mn+h),
\end{equation}
then $F\equiv0$.  The same conclusion holds if the right-hand side is
$\Lambda(mn+h)$.
\end{theorem}

\begin{proof}
Fix any residue pair $(a,b)\pmod Q$.  Choose distinct primes $r,s$ with
$(rs,Qh)=1$.  The Chinese remainder
theorem gives simultaneous solutions
\[
 m\equiv a\pmod Q,
 \quad n\equiv b\pmod Q,
 \quad m\equiv1\pmod{rs},
 \quad n\equiv-h\pmod{rs}.
\]
Adding multiples of $Qrs$ makes $m,n$ arbitrarily large and ensures
$mn+h>rs$.  The target is divisible by the two distinct primes $r,s$;
hence it is composite and is not a prime power.  Therefore both
$\one_{\Pp}(mn+h)$ and $\Lambda(mn+h)$ are zero.  Periodicity and
nonnegativity give
\[
 0\le F(a,b)=F(m,n)\le0.
\]
Since $(a,b)$ was arbitrary, $F$ vanishes identically.
\end{proof}

\begin{remark}[Precise scope]\label{rem:minorant-scope}
The theorem concerns a fixed period and a pointwise nonnegative minorant.  It
does not apply to signed identities, mean-square approximations, moduli
growing with the observation scale, or bilinear and dispersion estimates.
It says only that a fixed finite-state local observable cannot itself be the
positive target-prime detector required for a lower-bound sieve.
\end{remark}

\subsection{A CRT nullspace witness}

For a finite signed measure $\nu$ on $\N^2$, let
\begin{equation}\label{eq:crt-pushforward}
 \Pi_Q\nu=(\pi_Q)_\#\nu,
 \qquad
 \pi_Q(m,n)=(m\bmod Q,n\bmod Q).
\end{equation}
Every $Q$-local linear statistic factors through $\Pi_Q$.

Define the target-prime and target-parity functionals
\begin{align}
 L_{\rm prime}(\nu)
 &=\int\one_{\Pp}(mn+2)\,d\nu(m,n),\label{eq:prime-functional}\\
 L_{\lambda}(\nu)
 &=\int\lambda(mn+2)\,d\nu(m,n),\label{eq:liouville-functional}
\end{align}
where $\lambda(N)=(-1)^{\Omega(N)}$.

\begin{theorem}[Fixed-CRT non-identifiability]\label{thm:fixed-crt-nonidentifiability}
Let $Q\ge3$ be odd.  There exist two nonnegative unit point masses
$\nu_\square$ and $\nu_{\mathsf p}$ on $\N^2$ such that
\begin{equation}\label{eq:same-crt-data}
 \Pi_Q\nu_\square=\Pi_Q\nu_{\mathsf p},
\end{equation}
but
\begin{align}
 L_{\rm prime}(\nu_\square)&=0,&
 L_{\rm prime}(\nu_{\mathsf p})&=1,
 \label{eq:prime-functional-separation}\\
 L_\lambda(\nu_\square)&=+1,&
 L_\lambda(\nu_{\mathsf p})&=-1.
 \label{eq:lambda-functional-separation}
\end{align}
Both targets and both factor coordinates are units modulo $Q$.
Consequently the two measures have identical values for every fixed-$Q$ CRT
observable, every Dirichlet-character observable modulo a divisor of $Q$ on
the factor residues, and the centered target-rough kernel at the squarefree
modulus $\operatorname{rad}(Q)$.
\end{theorem}

\begin{proof}
Put
\[
 T_\square=(Q+1)^2,
 \qquad
 x_\square=(1,T_\square-2).
\]
Dirichlet's theorem supplies a prime $\ell\equiv1\pmod Q$, which may be
chosen arbitrarily large \cite[Chapter~6]{MontgomeryVaughan2007}.  Put
$x_{\mathsf p}=(1,\ell-2)$ and let the two
measures be the corresponding point masses.  Since both targets are
congruent to $1\pmod Q$,
\[
 x_\square\equiv x_{\mathsf p}\equiv(1,-1)\pmod Q.
\]
This proves \eqref{eq:same-crt-data} and the unit assertions.  The first
target is a square, so its Liouville value is $+1$ and it is composite; the
second is prime, so its Liouville value is $-1$.  This proves
\eqref{eq:prime-functional-separation}--\eqref{eq:lambda-functional-separation}.
Every stated local statistic is a function of $\pi_Q(m,n)$ and therefore has
the same integral against the two measures.
\end{proof}

Let $v=\nu_\square-\nu_{\mathsf p}$.  Then
\begin{equation}\label{eq:nullspace-witness}
 \Pi_Qv=0,
 \qquad
 L_{\rm prime}(v)=-1,
 \qquad
 L_\lambda(v)=2.
\end{equation}
Thus neither target functional belongs to the row space generated by all
fixed-$Q$ local observables.

\begin{corollary}[Minimax consequence]\label{cor:minimax-local-estimator}
Any estimator of $L_{\rm prime}$ which depends only on $\Pi_Q\nu$ has
worst-case absolute error at least $1/2$ on the two point masses in
\cref{thm:fixed-crt-nonidentifiability}.  For $L_\lambda$, the corresponding
lower bound is $1$.
\end{corollary}

\begin{proof}
The estimator must output the same number on two inputs with identical
pushforward.  A single real number cannot be within less than $1/2$ of both
$0$ and $1$, nor within less than $1$ of both $-1$ and $+1$.
\end{proof}

\subsection{An exact certificate at \texorpdfstring{$Q=385$}{Q=385}}

Take
\begin{equation}\label{eq:certificate-parameters}
 Q=5\cdot7\cdot11=385,
 \qquad h=2.
\end{equation}
The two target values in the certificate are
\begin{align}
 T_\square&=386^2=148996=2^2\cdot193^2,
 \label{eq:certificate-square}\\
 T_{\mathsf p}&=148996+385=149381.
 \label{eq:certificate-prime}
\end{align}
They are represented by
\begin{equation}\label{eq:certificate-atoms}
 x_\square=(1,148994),
 \qquad
 x_{\mathsf p}=(1,149379),
\end{equation}
and both reduce to $(1,384)$ modulo $385$.  Their common local signature is
\begin{equation}\label{eq:certificate-signature}
\begin{array}{c@{\qquad}ccc}
\toprule
p&m\bmod p&n\bmod p&(mn+2)\bmod p\\
\midrule
5&1&4&1\\
7&1&6&1\\
11&1&10&1\\
\bottomrule
\end{array}
\end{equation}
Every window prime therefore treats both targets as the same survivor.  The
local density and centered-kernel value are exactly
\begin{equation}\label{eq:certificate-kernel-values}
 \kappa_Q=\frac34\frac56\frac9{10}=\frac9{16},
 \qquad
 K_{Q,2}^{\circ}(x_\square)
 =K_{Q,2}^{\circ}(x_{\mathsf p})
 =\frac79.
\end{equation}

For completeness, primality of $149381$ is certified without probabilistic
testing.  Its predecessor factors completely as
\begin{equation}\label{eq:lucas-factorization}
 149381-1=149380=2^2\cdot5\cdot7\cdot11\cdot97.
\end{equation}
With witness $a=3$,
\begin{equation}\label{eq:lucas-main-congruence}
 3^{149380}\equiv1\pmod{149381},
\end{equation}
and for every $q\in\{2,5,7,11,97\}$,
\begin{equation}\label{eq:lucas-gcd-condition}
 \gcd\bigl(3^{149380/q}-1,149381\bigr)=1.
\end{equation}
The residues of $3^{149380/q}$ modulo $149381$, in the displayed order of
$q$, are
\begin{equation}\label{eq:lucas-residues}
 149380,\quad11332,\quad34444,\quad38519,\quad28976.
\end{equation}
The Lucas criterion and \eqref{eq:lucas-factorization} prove that $149381$
is prime.  Explicitly, the criterion says that if
$a^{N-1}\equiv1\pmod N$ and
$\gcd(a^{(N-1)/q}-1,N)=1$ for every prime $q\mid N-1$, then $N$ is prime;
\eqref{eq:lucas-main-congruence}--\eqref{eq:lucas-residues} verify exactly
these hypotheses.

fractions, functional gaps, and the Lucas certificate using only exact

certificate byte for byte; see \cref{sec:certificates-boundaries}.

\begin{remark}[Relation to the classical parity problem]\label{rem:finite-vs-classical-parity}
The square/prime witness has opposite Liouville parity and annihilates every
fixed-$Q$ observable, so it is an exact finite-information obstruction.  It
is not the classical sieve parity phenomenon.  For Selberg's limitation and
an explicit model sequence satisfying strong remainder axioms while
containing no prime mass, see \citet{Selberg1949,FriedlanderIwaniec1998}; for
a modern limitation within a specified Selberg-sieve framework, see
\citet{Polymath2014}.  Those results compare much richer asymptotic data and
specify which additional bilinear axioms can change the conclusion.  Our
theorem is intentionally narrower and is used only to delimit the finite
dynamical information of the present series.
\end{remark}
