\section{The determinant dictionary and bin-energy survival}
\label{sec:dictionary}

Even a positive pre-bin energy
\(\|\cB_Xz_X\|^2\) need not survive determinant aggregation.
The correct additional statistic is a lower modulus of the literal
binning map.

\begin{definition}[Verified reassembled subspace]
\label{def:reassembled-subspace}
Let \(E_X\subseteq\C^{\cY_X}\) be a subspace constructed from literal
constraints and satisfying
\[
 \cB_Xz_X\in E_X.
\]
Its determinant-binning lower modulus is
\begin{equation}
 s_X=s(\cJ_X;E_X)
 :=
 \inf_{\substack{u\in E_X\\u\ne0}}
 \frac{\|\cJ_Xu\|_2}{\|u\|_2}.
 \label{eq:lower-modulus}
\end{equation}
If \(E_X=\{0\}\), set \(s_X=1\).
\end{definition}

The convention for the zero subspace is immaterial because then the
reassembled vector is zero.  For a nonzero subspace,
\(s_X>0\) is equivalent to injectivity of
\(\cJ_X|_{E_X}\).  It is not implied by surjectivity of
\(\cJ_X\) on the ambient determinant space.

\begin{theorem}[Binning--dictionary lower bound]
\label{thm:dictionary-lower}
Assume the packet of \cref{def:complete-packet}, and suppose that for
some \(\Gamma_{{\rm pre},X}>0\) and
\(\delta_{{\rm reasm},X}\ge0\),
\begin{equation}
 \|\cB_Xz_X\|_2^2
 \ge
 \delta_{{\rm reasm},X}\Gamma_{{\rm pre},X}
 D_{{\rm raw},X}.
 \label{eq:prebin-survival}
\end{equation}
Let \(E_X\) be as in \cref{def:reassembled-subspace}.  Then
\begin{equation}
 \boxed{\displaystyle
 D_X\ge
 \left(
 s_X\sqrt{\delta_{{\rm reasm},X}\Gamma_{{\rm pre},X}}
 -\frac{\|r_X\|_2}{\sqrt{D_{{\rm raw},X}}}
 \right)_+^2D_{{\rm raw},X}.}
 \label{eq:dictionary-lower}
\end{equation}
\end{theorem}

\begin{proof}
By the definition of \(s_X\) and
\eqref{eq:prebin-survival},
\[
 \|\cJ_X\cB_Xz_X\|_2
 \ge s_X\|\cB_Xz_X\|_2
 \ge
 s_X\sqrt{\delta_{{\rm reasm},X}\Gamma_{{\rm pre},X}
 D_{{\rm raw},X}}.
\]
Since \(a_X=\cJ_X\cB_Xz_X+r_X\), the reverse-triangle
inequality gives
\[
 \|a_X\|_2
 \ge
 \left(
 s_X\sqrt{\delta_{{\rm reasm},X}\Gamma_{{\rm pre},X}
 D_{{\rm raw},X}}
 -\|r_X\|_2
 \right)_+.
\]
Square and use \(D_X=\|a_X\|_2^2\).
\end{proof}

\begin{corollary}[Exact dictionary]
\label{cor:exact-dictionary}
If \(r_X=0\), then
\begin{equation}
 D_X
 \ge
 s_X^2\delta_{{\rm reasm},X}
 \Gamma_{{\rm pre},X}D_{{\rm raw},X}.
 \label{eq:exact-dictionary-energy}
\end{equation}
\end{corollary}

\begin{definition}[Polynomial-loss ledger]
\label{def:polynomial-ledger}
Fix nonnegative exponents
\(\lambda_{\rm raw},\lambda_s,\lambda_\delta,\lambda_\Gamma,
\lambda_r\).  Suppose the following lower bounds hold:
\begin{align}
 D_{{\rm raw},X}
 &\ge X^{-\lambda_{\rm raw}+o(1)}Q_X^3J_X^2,
 \label{eq:raw-exponent}\\
 s_X
 &\ge X^{-\lambda_s+o(1)},
 \label{eq:s-exponent}\\
 \delta_{{\rm reasm},X}
 &\ge X^{-\lambda_\delta+o(1)},
 \label{eq:delta-exponent}\\
 \Gamma_{{\rm pre},X}
 &\ge X^{-\lambda_\Gamma+o(1)}.
 \label{eq:Gamma-exponent}
\end{align}
Write
\begin{equation}
 \varepsilon_{r,X}
 :=\frac{\|r_X\|_2}{\sqrt{D_{{\rm raw},X}}}.
 \label{eq:normalized-residual}
\end{equation}
If, for some \(0<\tau_X\le1\),
\begin{equation}
 \varepsilon_{r,X}
 \le
 (1-\tau_X)
 s_X\sqrt{\delta_{{\rm reasm},X}\Gamma_{{\rm pre},X}},
 \qquad
 \tau_X\ge X^{-\lambda_r+o(1)},
 \label{eq:residual-separation}
\end{equation}
then \(\lambda_r\) is called the dictionary-separation loss.
\end{definition}

\begin{corollary}[Explicit certified energy-loss exponent]
\label{cor:lambda-D}
Under \cref{def:polynomial-ledger},
\begin{equation}
 D_X
 \ge
 X^{-\lambda_D^{\rm cert}+o(1)}Q_X^3J_X^2,
 \qquad
 \boxed{
 \lambda_D^{\rm cert}
 =
 \lambda_{\rm raw}
 +2\lambda_s+\lambda_\delta+\lambda_\Gamma+2\lambda_r.}
 \label{eq:lambda-D-ledger}
\end{equation}
\end{corollary}

\begin{proof}
By \eqref{eq:residual-separation}, the positive part in
\eqref{eq:dictionary-lower} is at least
\[
 \tau_Xs_X
 \sqrt{\delta_{{\rm reasm},X}\Gamma_{{\rm pre},X}}.
\]
Insert \eqref{eq:raw-exponent}--\eqref{eq:Gamma-exponent}, square,
and collect exponents.
\end{proof}

\begin{remark}[A lower modulus is arithmetic data]
For the sum map \eqref{eq:literal-binning}, the ambient kernel is
usually large.  A useful \(s_X\) therefore requires a genuinely
restricted subspace \(E_X\), defined by identities satisfied by the
actual reassembled coefficient.  Choosing \(E_X\) after inspecting
only the target vector, or replacing it by a numerically fitted
one-dimensional span, does not prove a uniform asymptotic theorem.
The constraints defining \(E_X\) and the lower-modulus proof must
come from archived identities independently of the value of \(D_X\);
the proof of \(s_X\) may not invoke the determinant-energy lower bound
it is meant to establish.
\end{remark}

\begin{remark}[Possible arithmetic hardness]
Expanding \(D_X=\sum_n|A_{C_X}(n)|^2\) reintroduces pairs of literal
M\"obius-weighted atoms constrained to have equal determinant.  A
lower bound at the natural scale may therefore encode a signed
two-point correlation of Chowla type.  The algebra above identifies
that possibility; it does not remove it.  Averaged results for
multiplicative correlations provide important benchmarks
\citep{MatomakiRadziwillTao2015,Tao2016LogChowla}, but do not by
themselves establish this fixed-\(h_0\), determinant-fiber lower
bound.
\end{remark}
