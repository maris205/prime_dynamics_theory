\section{Integral normal form on a fixed-determinant line}
\label{sec:tpc33-affine-normal}

We isolate the arithmetic geometry used by every affine column.  Throughout
this section,
\begin{equation}
 h\in\Z\setminus\{0\},\qquad a,s\in\N,\qquad
 (a,s)=1,\qquad(as,h)=1.
 \label{eq:tpc33-affine-hypotheses}
\end{equation}

\subsection{The \texorpdfstring{\(SL_2(\Z)\)}{SL2(Z)} parametrization}

Choose \(x,y\in\Z\) with
\begin{equation}
 sx-ay=1.
 \label{eq:tpc33-bezout}
\end{equation}
Set
\begin{equation}
 V=\begin{pmatrix}x&a\\y&s\end{pmatrix}\in SL_2(\Z).
 \label{eq:tpc33-unimodular-matrix}
\end{equation}

\begin{theorem}[Fixed-determinant normal form]
\label{thm:tpc33-affine-normal-form}
Every integral solution of
\begin{equation}
 sU-aD=h
 \label{eq:tpc33-fixed-determinant-equation}
\end{equation}
has a unique representation
\begin{equation}
 \boxed{
 U_t=hx+at,
 \qquad
 D_t=hy+st,
 \qquad t\in\Z.}
 \label{eq:tpc33-canonical-affine-forms}
\end{equation}
The inverse coordinates are
\begin{equation}
 \boxed{
 \binom ht
 =\begin{pmatrix}s&-a\\-y&x\end{pmatrix}
  \binom UD.}
 \label{eq:tpc33-affine-inverse}
\end{equation}
For every \(t\), with gcds understood positively,
\begin{equation}
 \boxed{
 (D_t,U_t)=(D_t,h)=(U_t,h)=(h,t).}
 \label{eq:tpc33-gcd-identities}
\end{equation}
Moreover,
\begin{equation}
 (D_t,s)=1,
 \qquad
 (U_t,a)=1.
 \label{eq:tpc33-slope-coprimality}
\end{equation}
\end{theorem}

\begin{proof}
The determinant of \(V\) is one.  Multiplying
\(V(h,t)^{\mathsf T}\) gives \eqref{eq:tpc33-canonical-affine-forms},
and \(sU_t-aD_t=(sx-ay)h=h\).  Conversely, the inverse matrix in
\eqref{eq:tpc33-affine-inverse} is integral.  Its first coordinate is
\(sU-aD=h\), so every solution has one and only one integral second
coordinate \(t\).

An integral unimodular transformation preserves the ideal generated by the
two coordinates.  Hence
\((D_t,U_t)=(h,t)\).  Also
\[
 (D_t,h)=(st,h)=(t,h),
 \qquad
 (U_t,h)=(at,h)=(t,h),
\]
because \((as,h)=1\).  This proves \eqref{eq:tpc33-gcd-identities}.
Finally, \eqref{eq:tpc33-bezout} implies \((y,s)=(x,a)=1\).  Therefore
\[
 (D_t,s)=(hy,s)=1,
 \qquad
 (U_t,a)=(hx,a)=1,
\]
again using \((as,h)=1\).
\end{proof}

Every other B\'ezout pair is
\begin{equation}
 x'=x+ak,\qquad y'=y+sk,\qquad k\in\Z.
 \label{eq:tpc33-bezout-change}
\end{equation}
For a fixed physical solution, changing from \((x,y)\) to \((x',y')\)
changes its affine coordinate from \(t\) to \(t-hk\).  Thus the residue
class \(t\pmod h\), and in particular \((t,h)\), is intrinsic even though
the integer origin is not.

\begin{corollary}[Primitive layer]
\label{cor:tpc33-primitive-layer}
The following conditions are equivalent:
\begin{equation}
 (D_t,h)=1,\qquad
 (U_t,h)=1,\qquad
 (D_t,U_t)=1,\qquad
 (t,h)=1.
 \label{eq:tpc33-primitive-equivalences}
\end{equation}
\end{corollary}

\begin{proof}
All four gcds are equal by \eqref{eq:tpc33-gcd-identities}.
\end{proof}

\subsection{M\"obius layers and the quadratic discriminant}

Let
\begin{equation}
 g=(h,t)=(D_t,U_t),\qquad
 D_t=gD_t',\qquad U_t=gU_t'.
 \label{eq:tpc33-common-gcd-layer}
\end{equation}
Then \((D_t',U_t')=1\).

\begin{proposition}[Exact M\"obius gcd layer]
\label{prop:tpc33-mobius-gcd-layer}
Whenever \(D_t,U_t>0\),
\begin{equation}
 \boxed{
 \mu(D_t)\mu(U_t)
 =\mu^2(g)\one_{(g,D_t'U_t')=1}\mu(D_t'U_t').}
 \label{eq:tpc33-mobius-gcd-layer}
\end{equation}
In particular, on the primitive layer \((t,h)=1\),
\begin{equation}
 \boxed{
 \mu(D_t)\mu(U_t)=\mu(D_tU_t).}
 \label{eq:tpc33-primitive-mobius-fusion}
\end{equation}
\end{proposition}

\begin{proof}
If \(g\) is not squarefree or shares a prime with \(D_t'U_t'\), both sides
of \eqref{eq:tpc33-mobius-gcd-layer} vanish.  Otherwise \(g,D_t',U_t'\)
are pairwise coprime and
\[
 \mu(gD_t')\mu(gU_t')
 =\mu(g)^2\mu(D_t')\mu(U_t')
 =\mu^2(g)\mu(D_t'U_t').
\]
The primitive statement follows from \(g=1\).
\end{proof}

The fused polynomial is
\begin{equation}
 F_{a,s,h}(t)
 :=D_tU_t
 =as t^2+h(sx+ay)t+h^2xy.
 \label{eq:tpc33-reducible-quadratic}
\end{equation}

\begin{proposition}[Square discriminant]
\label{prop:tpc33-square-discriminant}
The discriminant of \(F_{a,s,h}\) is exactly
\begin{equation}
 \boxed{\operatorname{disc}F_{a,s,h}=h^2.}
 \label{eq:tpc33-square-discriminant}
\end{equation}
\end{proposition}

\begin{proof}
By \eqref{eq:tpc33-bezout},
\begin{align*}
 \operatorname{disc}F_{a,s,h}
 &=h^2\bigl((sx+ay)^2-4asxy\bigr)\\
 &=h^2(sx-ay)^2=h^2.
\end{align*}
\end{proof}

Thus the two affine signs form a reducible quadratic M\"obius sequence with
a fixed square discriminant.  The identity is structural, not a
cancellation theorem: the case \(a=s=h=1\) contains the open sequence
\(\mu(t)\mu(t+1)\), as recorded in
\cref{sec:tpc33-sharp-obstructions}.

\subsection{The reduced physical origin}

For column disintegration it is often more convenient to choose the unique
integer \(D_0\) with
\begin{equation}
 1\le D_0\le s,\qquad aD_0+h\equiv0\pmod s,
 \label{eq:tpc33-reduced-origin-D}
\end{equation}
and put
\begin{equation}
 U_0=\frac{aD_0+h}{s}.
 \label{eq:tpc33-reduced-origin-U}
\end{equation}
Then all solutions are
\begin{equation}
 D_r=D_0+sr,\qquad U_r=U_0+ar,
 \label{eq:tpc33-reduced-origin-forms}
\end{equation}
with
\begin{equation}
 |D_0|\le s,\qquad |U_0|\le a+|h|.
 \label{eq:tpc33-reduced-intercept-bounds}
\end{equation}
This origin differs from the canonical one by an integral translation.
Accordingly, \((D_r,h)=1\) excludes one translated residue class of
\(r\pmod p\) for every \(p\mid h\); it should not in general be replaced
by the literal condition \((r,h)=1\).
