\section{A sharp bounded dyadic counterexample}
\label{sec:dyadic-counterexample}

\begin{definition}[Alternating dyadic blocks]
\label{def:dyadic-sequence}
For \(n\ge1\), define
\begin{equation}
 a_n=(-1)^k\qquad\text{when }2^k\le n<2^{k+1}.
\label{eq:dyadic-sequence}
\end{equation}
Thus \(|a_n|=1\).
\end{definition}

Write
\begin{equation}
 h_k:=\sum_{n=2^k}^{2^{k+1}-1}\frac1n.
\label{eq:dyadic-harmonic-mass}
\end{equation}
Integral comparison gives
\begin{equation}
 \log2\le h_k<\log2+2^{-k}.
\label{eq:dyadic-harmonic-estimate}
\end{equation}

\begin{theorem}[Bounded logarithmic primitive, oscillating natural mean]
\label{thm:dyadic-counterexample}
For the sequence \eqref{eq:dyadic-sequence}:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=2.5em]
 \item \(L_N=O(1)\);
 \item \(\lambda_N=L_N/H_N\to0\);
 \item for \(N_K=2^{K+1}-1\),
 \begin{equation}
  S_{N_K}=\frac{1-(-2)^{K+1}}3,
 \label{eq:dyadic-natural-sum}
 \end{equation}
 and hence
 \begin{equation}
  \frac{S_{N_K}}{N_K}\longrightarrow
  \begin{cases}
   1/3,&K\ \text{even},\\
   -1/3,&K\ \text{odd}.
  \end{cases}
 \label{eq:dyadic-natural-limits}
 \end{equation}
\end{enumerate}
In particular, boundedness of \(L_N\), and even convergence of the
normalized logarithmic mean to zero, does not imply ordinary
cancellation.
\end{theorem}

\begin{proof}
If \(2^K\le N<2^{K+1}\), then
\[
 L_N=
 \sum_{k=0}^{K-1}(-1)^kh_k
 +(-1)^K\sum_{n=2^K}^{N}\frac1n.
\]
Write \(h_k=\log2+e_k\), where
\(0\le e_k<2^{-k}\).  The alternating sum of the constant
\(\log2\) terms is bounded, the error series is absolutely
summable, and the final partial block has bounded harmonic mass.
This proves \(L_N=O(1)\).  Since \(H_N\to\infty\), it also proves
\(\lambda_N\to0\).

At the right endpoint of the \(K\)-th complete block,
\[
 S_{N_K}=\sum_{k=0}^K(-1)^k2^k
 =\frac{1-(-2)^{K+1}}3.
\]
Dividing by \(2^{K+1}-1\) gives
\eqref{eq:dyadic-natural-limits}.
\end{proof}

\begin{theorem}[Uniform Tao-shaped logarithmic windows]
\label{thm:uniform-log-windows}
There is an absolute constant \(K_0\) such that, for every
\(x\ge2\) and \(1\le\Omega\le x\),
\begin{equation}
 \left|
 \sum_{x/\Omega<n\le x}\frac{a_n}{n}
 \right|\le K_0.
\label{eq:uniform-window-bound}
\end{equation}
Consequently, for every function
\(1\le\omega(x)\le x\) with \(\omega(x)\to\infty\),
\begin{equation}
 \frac1{\log\omega(x)}
 \sum_{x/\omega(x)<n\le x}\frac{a_n}{n}
 \longrightarrow0.
\label{eq:normalized-window-cancellation}
\end{equation}
\end{theorem}

\begin{proof}
The interval \((x/\Omega,x]\) meets at most two dyadic blocks
partially.  Each partial block has harmonic mass at most
\(\log2+1\).  The remaining complete blocks form a consecutive
range.  By \eqref{eq:dyadic-harmonic-estimate}, their contribution
is an alternating sum of copies of \(\log2\), of absolute value at
most \(\log2\), plus an absolutely summable error of total size at
most \(2\).  Thus \eqref{eq:uniform-window-bound} holds, for example
with \(K_0=8\).  Divide by \(\log\omega(x)\to\infty\).
\end{proof}

\begin{corollary}[Sharp black-box failure]
\label{cor:black-box-failure}
For bounded sequences in general, even normalized logarithmic
cancellation on every growing terminal multiplicative window does
not imply \(S_N/N\to0\).
\end{corollary}

\begin{proof}
Combine \cref{thm:dyadic-counterexample,thm:uniform-log-windows}.
\end{proof}

\begin{remark}[Why this does not contradict the positive criteria]
\label{rem:counterexample-local-failure}
At the end of a dyadic block, the terminal half is essentially one
constant-sign block, so \(\omega_N(L)\) stays bounded below by a
positive constant close to \(\log2\).  Also
\[
 V_k=\sum_{2^k\le n<2^{k+1}}\frac1n=h_k\to\log2.
\]
Thus the counterexample fails both additional hypotheses in
\cref{sec:local-variation}.
\end{remark}

\begin{remark}[Scope]
\label{rem:counterexample-scope}
The sequence \eqref{eq:dyadic-sequence} is not multiplicative.  It
proves a logical nonimplication for structure-free Tauberian
arguments.  It neither models the literal M\"obius coefficient nor
rules out a conversion using multiplicativity, entropy decrement,
pretentious distance, or another arithmetic input.
\end{remark}
